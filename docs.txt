CREATE {
    это команда для создания новой таблицы
}
INSERT {
    это команда добавляет данные в таблицу и при добавлении данных в каждый столбец таблицы не требуется указывать названия столбцов.
}
UPDATE {
    это каманда для обновления данных таблиц.
}
SELECT {
    доет восможность получить данные из выбранной таблицы
}
DROP {
    используют для удаления данных с помощью запроса
}
